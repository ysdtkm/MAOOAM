148 FUNCTION gasdev(idum)
149  INTEGER :: idum
150  REAL(KIND=8) :: gasdev,ran2
151  ! USES ran2
152  INTEGER :: iset
153  REAL(KIND=8) :: fac,gset,rsq,v1,v2
154  SAVE iset,gset
155  DATA iset/0/
156  if (idum.lt.0) iset=0
157  if (iset.eq.0) then
158 1 v1=2.d0*ran2(idum)-1.
159  v2=2.d0*ran2(idum)-1.
160  rsq=v1**2+v2**2
161  if (rsq.ge.1.d0.or.rsq.eq.0.d0) goto 1
162  fac=sqrt(-2.*log(rsq)/rsq)
163  gset=v1*fac
164  gasdev=v2*fac
165  iset=1
166  else
167  gasdev=gset
168  iset=0
169  endif
170  return
171 END FUNCTION gasdev
